// Wake Lockオブジェクトを保持する変数 let wakeLock = null; // 画面をスリープさせないための関数 async function requestWakeLock() { try { // 画面の「スリープ抑制」をリクエスト wakeLock = await navigator.wakeLock.request('screen'); console.log('スクリーンセーバーの起動を防止しました'); // タブが切り替わったり最小化された後に復帰した際、再度ロックをかけるための処理 wakeLock.addEventListener('release', () => { console.log('スリープ抑制が解除されました'); }); } catch (err) { console.error(`${err.name}, ${err.message}`); } } // 先ほどの「スタートボタン」のイベントリスナー内にこれを追加します startButton.addEventListener('click', () => { // 既存のセンサー許可処理... // スリープ防止をリクエスト requestWakeLock(); // ゲーム開始処理... }); // アプリがバックグラウンドから戻ったときに再取得する処理 document.addEventListener('visibilitychange', async () => { if (wakeLock !== null && document.visibilityState === 'visible') { await requestWakeLock(); } });
